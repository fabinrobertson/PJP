  vename    EMP.ename%TYPE;
  vsalary   EMP.salary%TYPE;
  CURSOR c1 IS 
  SELECT ename, salary FROM EMP ORDER BY salary DESC;
BEGIN
  n := &number;
  OPEN c1;
  LOOP
  FETCH c1 INTO vename, vsalary;
  EXIT WHEN c1%ROWCOUNT > n;
  dbms_output.put_line(vename||" "||vsalary)
  END LOOP;
  CLOSE c1;
END;
